63 static inline void handle_interrupts(timer16_Sequence_t timer, volatile uint16_t *TCNTn, volatile uint16_t* OCRnA)
64 {
65  if( Channel[timer] < 0 )
66  *TCNTn = 0; // channel set to -1 indicated that refresh interval completed so reset the timer
67  else{
68  if( SERVO_INDEX(timer,Channel[timer]) < ServoCount && SERVO(timer,Channel[timer]).Pin.isActive == true )
69  digitalWrite( SERVO(timer,Channel[timer]).Pin.nbr,LOW); // pulse this channel low if activated
70  }
71 
72  Channel[timer]++; // increment to the next channel
73  if( SERVO_INDEX(timer,Channel[timer]) < ServoCount && Channel[timer] < SERVOS_PER_TIMER) {
74  *OCRnA = *TCNTn + SERVO(timer,Channel[timer]).ticks;
75  if(SERVO(timer,Channel[timer]).Pin.isActive == true) // check if activated
76  digitalWrite( SERVO(timer,Channel[timer]).Pin.nbr,HIGH); // its an active channel so pulse it high
77  }
78  else {
79  // finished all channels so wait for the refresh period to expire before starting over
80  if( ((unsigned)*TCNTn) + 4 < usToTicks(REFRESH_INTERVAL) ) // allow a few ticks to ensure the next OCR1A not missed
81  *OCRnA = (unsigned int)usToTicks(REFRESH_INTERVAL);
82  else
83  *OCRnA = *TCNTn + 4; // at least REFRESH_INTERVAL has elapsed
84  Channel[timer] = -1; // this will get incremented at the end of the refresh period to start again at the first channel
85  }
86 }

111 static void initISR(timer16_Sequence_t timer) {
112  if(timer == _timer1) {
113  TCCR1A = 0; // normal counting mode
