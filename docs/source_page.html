145  uint8_t twi_readFrom(uint8_t address, uint8_t* data, uint8_t length, uint8_t sendStop) {
146  uint8_t i;
147  if(TWI_BUFFER_LENGTH < length) return 0;
148  while(TWI_READY != twi_state) continue;
149  twi_state = TWI_MRX;
150  twi_sendStop = sendStop;
151  twi_error = 0xFF;
152  twi_masterBufferIndex = 0;
153  twi_masterBufferLength = length-1;
154  twi_slarw = TW_READ;
155  twi_slarw |= address << 1;
156  if (true == twi_inRepStart) {
157  twi_inRepStart = false;
158  TWDR = twi_slarw;
159  TWCR = _BV(TWINT) | _BV(TWEA) | _BV(TWEN) | _BV(TWIE); // enable INTs, but not START
160  }
161  else
162  TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA) | _BV(TWINT) | _BV(TWSTA);
163  while(TWI_MRX == twi_state) continue;
164  if (twi_masterBufferIndex < length)
165  length = twi_masterBufferIndex;
166  for(i = 0; i < length; ++i) data[i] = twi_masterBuffer[i];
167  return length;
168  }
169  uint8_t twi_writeTo(uint8_t address, uint8_t* data, uint8_t length, uint8_t wait, uint8_t sendStop) {
170  uint8_t i;
171  if(TWI_BUFFER_LENGTH < length) return 1;
172  while(TWI_READY != twi_state) continue;
173  twi_state = TWI_MTX;
174  twi_sendStop = sendStop;
175  twi_error = 0xFF;
176  twi_masterBufferIndex = 0;
177  twi_masterBufferLength = length;
178  for(i = 0; i < length; ++i) twi_masterBuffer[i] = data[i];
179  twi_slarw = TW_WRITE;
180  twi_slarw |= address << 1;
181  if (true == twi_inRepStart) {
182  twi_inRepStart = false;
183  TWDR = twi_slarw;
184  TWCR = _BV(TWINT) | _BV(TWEA) | _BV(TWEN) | _BV(TWIE); // enable INTs, but not START
185  }
186  else
187  TWCR = _BV(TWINT) | _BV(TWEA) | _BV(TWEN) | _BV(TWIE) | _BV(TWSTA); // enable INTs
188  while(wait && (TWI_MTX == twi_state)) continue;
189  if (twi_error == 0xFF) return 0;
190  else if (twi_error == TW_MT_SLA_NACK) return 2;
191  else if (twi_error == TW_MT_DATA_NACK) return 3;
192  else return 4;
193  }

266  else {

270  }

272  case TW_MR_SLA_NACK:
273  twi_stop();
274  break;

276  case TW_SR_GCALL_ACK:
277  case TW_SR_ARB_LOST_SLA_ACK:
278  case TW_SR_ARB_LOST_GCALL_ACK:
279  twi_state = TWI_SRX;
280  twi_rxBufferIndex = 0;
281  twi_reply(1);
282  break;
283  case TW_SR_DATA_ACK:
284  case TW_SR_GCALL_DATA_ACK:
285  if(twi_rxBufferIndex < TWI_BUFFER_LENGTH){
286  twi_rxBuffer[twi_rxBufferIndex++] = TWDR;
287  twi_reply(1);
288  } else twi_reply(0);
289  break;
290  case TW_SR_STOP:
291  if(twi_rxBufferIndex < TWI_BUFFER_LENGTH) twi_rxBuffer[twi_rxBufferIndex] = '\0';
292  twi_stop();
293  twi_onSlaveReceive(twi_rxBuffer, twi_rxBufferIndex);
294  twi_rxBufferIndex = 0;
295  twi_releaseBus();
296  break;
297  case TW_SR_DATA_NACK:
298  case TW_SR_GCALL_DATA_NACK:
299  twi_reply(0);
300  break;
301  case TW_ST_SLA_ACK:
302  case TW_ST_ARB_LOST_SLA_ACK:
303  twi_state = TWI_STX;
304  twi_txBufferIndex = 0;
305  twi_txBufferLength = 0;
306  twi_onSlaveTransmit();
307  if(0 == twi_txBufferLength){
308  twi_txBufferLength = 1;
309  twi_txBuffer[0] = 0x00;
310  }
311  case TW_ST_DATA_ACK:
312  TWDR = twi_txBuffer[twi_txBufferIndex++];
313  if(twi_txBufferIndex < twi_txBufferLength) twi_reply(1);
314  else twi_reply(0);
315  break;
316  case TW_ST_DATA_NACK:
317  case TW_ST_LAST_DATA:
318  twi_reply(1);
319  twi_state = TWI_READY;
320  break;
321  case TW_NO_INFO:
322  break;
323  case TW_BUS_ERROR:
324  twi_error = TW_BUS_ERROR;
325  twi_stop();
326  break;
327  }
