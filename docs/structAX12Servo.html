[COPY&PASTE] You can use this class like this:

DynamixelBoard servoBoard(board);
AX12Servo Links(servoBoard,1);
AX12Servo Rechts(servoBoard,2);
int main() {
Links.setPosition(50);
Rechts.setPosition(10);
while(true) {}
}
